CREATE DATABASE aplicacion_musica;
USE aplicacion_musica;

-- 1. Artistas
CREATE TABLE Artistas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    pais VARCHAR(50)
);

-- 2. Álbumes
CREATE TABLE Albumes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(100),
    id_artista INT,
    fecha_lanzamiento DATE,
    FOREIGN KEY (id_artista) REFERENCES Artistas(id)
);

-- 3. Géneros
CREATE TABLE Generos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50)
);

-- 4. Canciones
CREATE TABLE Canciones (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(100),
    duracion TIME,
    id_album INT,
    id_genero INT,
    id_productor INT,
    id_estudio INT,
    FOREIGN KEY (id_album) REFERENCES Albumes(id),
    FOREIGN KEY (id_genero) REFERENCES Generos(id),
    FOREIGN KEY (id_productor) REFERENCES Productores(id),
    FOREIGN KEY (id_estudio) REFERENCES EstudiosGrabacion(id)
);

-- 5. Productores
CREATE TABLE Productores (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    experiencia_anios INT
);

-- 6. Estudios de Grabación
CREATE TABLE EstudiosGrabacion (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    ciudad VARCHAR(100),
    pais VARCHAR(50)
);

-- 7. Proveedores (de CDs, vinilos u otros formatos físicos)
CREATE TABLE Proveedores (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    contacto VARCHAR(100),
    telefono VARCHAR(20)
);

-- 8. Inventario
CREATE TABLE Inventario (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_album INT,
    id_proveedor INT,
    stock INT,
    fecha_entrada DATE,
    FOREIGN KEY (id_album) REFERENCES Albumes(id),
    FOREIGN KEY (id_proveedor) REFERENCES Proveedores(id)
);

-- 9. Ventas
CREATE TABLE Ventas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_empleado INT,
    fecha DATE,
    total DECIMAL(10,2),
    FOREIGN KEY (id_empleado) REFERENCES Empleados(id)
);

-- 10. Empleados
CREATE TABLE Empleados (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    puesto VARCHAR(50),
    email VARCHAR(100)
);
