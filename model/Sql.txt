CREATE DATABASE tienda_videojuegos;
USE tienda_videojuegos;

-- 1. Clientes
CREATE TABLE Clientes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    email VARCHAR(100),
    telefono VARCHAR(20),
    direccion VARCHAR(200)
);

-- 2. Videojuegos
CREATE TABLE Videojuegos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(100),
    id_categoria INT,
    id_plataforma INT,
    id_desarrollador INT,
    precio DECIMAL(10,2),
    fecha_lanzamiento DATE,
    FOREIGN KEY (id_categoria) REFERENCES Categorías(id),
    FOREIGN KEY (id_plataforma) REFERENCES Plataformas(id),
    FOREIGN KEY (id_desarrollador) REFERENCES Desarrolladores(id)
);

-- 3. Plataformas
CREATE TABLE Plataformas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50)
);

-- 4. Categorías
CREATE TABLE Categorías (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50)
);

-- 5. Desarrolladores
CREATE TABLE Desarrolladores (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    pais VARCHAR(50)
);

-- 6. Proveedores
CREATE TABLE Proveedores (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    contacto VARCHAR(100),
    telefono VARCHAR(20)
);

-- 7. Inventario
CREATE TABLE Inventario (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_videojuego INT,
    id_proveedor INT,
    stock INT,
    fecha_entrada DATE,
    FOREIGN KEY (id_videojuego) REFERENCES Videojuegos(id),
    FOREIGN KEY (id_proveedor) REFERENCES Proveedores(id)
);

-- 8. Ventas
CREATE TABLE Ventas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_cliente INT,
    id_empleado INT,
    fecha DATE,
    total DECIMAL(10,2),
    FOREIGN KEY (id_cliente) REFERENCES Clientes(id),
    FOREIGN KEY (id_empleado) REFERENCES Empleados(id)
);

-- 9. DetallesVenta
CREATE TABLE DetallesVenta (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_venta INT,
    id_videojuego INT,
    cantidad INT,
    precio_unitario DECIMAL(10,2),
    FOREIGN KEY (id_venta) REFERENCES Ventas(id),
    FOREIGN KEY (id_videojuego) REFERENCES Videojuegos(id)
);

-- 10. Empleados
CREATE TABLE Empleados (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    puesto VARCHAR(50),
    email VARCHAR(100)
);
